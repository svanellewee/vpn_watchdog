#+BEGIN_SRC bash

  scutil --nc start "$VPNName"

  sleep 2.7
  osascript -e "tell application \"System Events\" to keystroke \"$PASSWORD\""
  osascript -e "tell application \"System Events\" to keystroke return"
  sleep 2

#+END_SRC
